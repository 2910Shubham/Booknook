-- =====================
-- BookNook Database Schema
-- =====================

-- Books table
CREATE TABLE IF NOT EXISTS books (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id       TEXT NOT NULL,
  title         TEXT NOT NULL,
  file_name     TEXT NOT NULL,
  file_path     TEXT,
  file_size     INTEGER,
  total_pages   INTEGER DEFAULT 0,
  cover_color   TEXT DEFAULT '#c8965a',
  uploaded_at   TIMESTAMPTZ DEFAULT NOW(),
  last_opened   TIMESTAMPTZ
);

-- Reading Progress
CREATE TABLE IF NOT EXISTS reading_progress (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id         TEXT NOT NULL,
  book_id         UUID REFERENCES books(id) ON DELETE CASCADE,
  current_page    INTEGER DEFAULT 1,
  total_pages     INTEGER DEFAULT 0,
  last_read_at    TIMESTAMPTZ DEFAULT NOW(),
  total_time_read INTEGER DEFAULT 0,
  UNIQUE(user_id, book_id)
);

-- Annotations
CREATE TABLE IF NOT EXISTS annotations (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id     TEXT NOT NULL,
  book_id     UUID REFERENCES books(id) ON DELETE CASCADE,
  page_number INTEGER NOT NULL,
  tool        TEXT NOT NULL CHECK (
                tool IN ('highlight','pen','underline','note')
              ),
  color       TEXT,
  data        JSONB NOT NULL DEFAULT '{}',
  created_at  TIMESTAMPTZ DEFAULT NOW(),
  updated_at  TIMESTAMPTZ DEFAULT NOW()
);

-- Bookmarks
CREATE TABLE IF NOT EXISTS bookmarks (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id     TEXT NOT NULL,
  book_id     UUID REFERENCES books(id) ON DELETE CASCADE,
  page_number INTEGER NOT NULL,
  label       TEXT DEFAULT '',
  created_at  TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(user_id, book_id, page_number)
);

-- Reading Sessions (daily tracking)
CREATE TABLE IF NOT EXISTS reading_sessions (
  id         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id    TEXT NOT NULL,
  book_id    UUID REFERENCES books(id) ON DELETE CASCADE,
  date       DATE DEFAULT CURRENT_DATE,
  duration   INTEGER DEFAULT 0,
  pages_read INTEGER DEFAULT 0,
  UNIQUE(user_id, book_id, date)
);

-- Reading Goals
CREATE TABLE IF NOT EXISTS reading_goals (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id       TEXT NOT NULL,
  book_id       UUID REFERENCES books(id) ON DELETE CASCADE,
  daily_minutes INTEGER DEFAULT 30,
  target_date   DATE,
  created_at    TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(user_id, book_id)
);

-- Indexes
CREATE INDEX IF NOT EXISTS idx_books_user_id
  ON books(user_id);
CREATE INDEX IF NOT EXISTS idx_annotations_book_page
  ON annotations(book_id, page_number);
CREATE INDEX IF NOT EXISTS idx_progress_user_book
  ON reading_progress(user_id, book_id);
CREATE INDEX IF NOT EXISTS idx_sessions_user_date
  ON reading_sessions(user_id, date);
CREATE INDEX IF NOT EXISTS idx_bookmarks_user_book
  ON bookmarks(user_id, book_id);

-- Row Level Security
ALTER TABLE books             ENABLE ROW LEVEL SECURITY;
ALTER TABLE reading_progress  ENABLE ROW LEVEL SECURITY;
ALTER TABLE annotations       ENABLE ROW LEVEL SECURITY;
ALTER TABLE bookmarks         ENABLE ROW LEVEL SECURITY;
ALTER TABLE reading_sessions  ENABLE ROW LEVEL SECURITY;
ALTER TABLE reading_goals     ENABLE ROW LEVEL SECURITY;

-- RLS Policies (Clerk JWT sub = user_id)
CREATE POLICY "users_own_books"
  ON books FOR ALL
  USING (user_id = current_setting('request.jwt.claims',true)::json->>'sub');

CREATE POLICY "users_own_progress"
  ON reading_progress FOR ALL
  USING (user_id = current_setting('request.jwt.claims',true)::json->>'sub');

CREATE POLICY "users_own_annotations"
  ON annotations FOR ALL
  USING (user_id = current_setting('request.jwt.claims',true)::json->>'sub');

CREATE POLICY "users_own_bookmarks"
  ON bookmarks FOR ALL
  USING (user_id = current_setting('request.jwt.claims',true)::json->>'sub');

CREATE POLICY "users_own_sessions"
  ON reading_sessions FOR ALL
  USING (user_id = current_setting('request.jwt.claims',true)::json->>'sub');

CREATE POLICY "users_own_goals"
  ON reading_goals FOR ALL
  USING (user_id = current_setting('request.jwt.claims',true)::json->>'sub');

-- Storage bucket for PDF files
INSERT INTO storage.buckets (id, name, public, file_size_limit)
VALUES ('books', 'books', false, 52428800)
ON CONFLICT DO NOTHING;

-- Storage RLS
CREATE POLICY "users_own_files"
  ON storage.objects FOR ALL
  USING (
    bucket_id = 'books'
    AND (storage.foldername(name))[1] =
      current_setting('request.jwt.claims',true)::json->>'sub'
  );

const isPublicRoute = createRouteMatcher([
  '/',
  '/sign-in(.*)',
  '/sign-up(.*)',
  '/api/webhook(.*)'

export default clerkMiddleware(async (auth, request) => {
  if (!isPublicRoute(request)) {
    await auth.protect()

STEP 4.3 — Annotation Sync:

Replace localStorage-only annotation saves with DB sync.

In useAnnotationPersistence.ts:

ON PAGE LOAD:
  1. Check localStorage cache first (instant render)
  2. Fetch GET /api/annotations?bookId=X&page=Y from DB
  3. Merge DB annotations into store (DB is source of truth)
  4. Update localStorage cache

ON ANNOTATION CREATE:
  1. Add to local store immediately (optimistic)
  2. POST /api/annotations
  3. On success: update annotation id in store with DB id
  4. On failure: add to offline queue, keep local copy

ON ANNOTATION DELETE:
  1. Remove from local store immediately (optimistic)
  2. DELETE /api/annotations/{id}
  3. On failure: restore annotation, add delete to offline queue

Create app/api/annotations/route.ts:
  GET ?bookId=&page=  → returns annotations array
  POST               → body: full annotation object
                     → INSERT, return with id

app/api/annotations/[id]/route.ts:
  DELETE → removes annotation (verify ownership first)

STEP 4.4 — Session Tracking:

In reader layout or page component:

On mount:
  sessionStartTime = Date.now()
  sessionStartPage = currentPage

On unmount + on document visibility hidden:
  duration = Math.floor((Date.now() - sessionStartTime) / 1000)
  pagesRead = Math.abs(currentPage - sessionStartPage)
  if (duration > 10) {  // Only log if read for > 10 seconds
    POST /api/sessions with { bookId, duration, pagesRead }
  }

Create app/api/sessions/route.ts:
  POST → body: { bookId, duration, pagesRead }
       → UPSERT to reading_sessions
       → If row for today exists: duration += new duration
       → If not: INSERT new row

STEP 4.5 — Offline Queue:

Create /lib/offlineQueue.ts:

QUEUE_KEY = 'booknook_offline_queue'

type QueuedOperation = {
  id: string
  type: 'create_annotation' | 'delete_annotation' | 'update_progress' | 'log_session'
  payload: object
  createdAt: number
  attempts: number
}

Functions:
  enqueue(operation): push to localStorage queue array
  processQueue(): 
    - Read all queued operations
    - For each: attempt the API call
    - On success: remove from queue
    - On failure: increment attempts, keep in queue (max 5 attempts)
    - Log results

Auto-trigger processQueue():
  - On app load (in root layout useEffect)
  - On window 'online' event

Show sync banner in reader:
  - Green dot + "Saved" when queue is empty
  - Amber spinning dot + "Saving..." when request in flight
  - Red dot + "Offline" when navigator.onLine is false
  - Tap red dot shows count of pending operations
